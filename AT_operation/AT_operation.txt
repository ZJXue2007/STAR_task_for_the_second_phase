#include <SoftwareSerial.h>
SoftwareSerial mybluetooth(2,3);//使用软串口，不用在上传数据的时候拔掉TX、RX
void setup() {
  Serial.begin(38400);
  mybluetooth.begin(38400);//HC-05在AT模式下的波特率
}
char data1,data2;
void loop() {//巧妙利用loop函数，通过循环读一个字符传一个，直到读完，然后一一接收传回来的字符，一次性读完使用readStringUntil则不够简洁
  if(Serial.available()>0){
    data1=Serial.read();
    mybluetooth.write(data1);
  }
  if(mybluetooth.available()>0){
    data2=mybluetooth.read();
    Serial.write(data2);
  }
}
