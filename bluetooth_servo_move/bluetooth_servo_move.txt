#include <Servo.h>
Servo anewservo;
void setup() {
  Serial.begin(9600);//HC-05在通信模式下波特率为9600
  anewservo.attach(9);
  Serial.println("系统已就绪。");
}
int angle;
void loop() {
  Serial.println("请输入希望旋转到的角度：");
  while(Serial.available()==0){//等待输入

  }
  if(Serial.available()>0){
    angle=Serial.parseInt();//把数据读进去
    if(angle>=0&&angle<=180){
      anewservo.write(angle);
      Serial.print("已转到");
      Serial.print(angle);
      Serial.println("°");//可以查看数据确认运行正常
    }else{
      Serial.println("您的角度超出范围，请重新输入。");
    }
  while(Serial.available()>0){//丢弃多余数据防止干扰
    Serial.read();
  }
}
}
